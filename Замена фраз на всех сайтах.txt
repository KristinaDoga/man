CREATE TABLE word_replace (
    id INT NOT NULL AUTO_INCREMENT,
    data VARCHAR(255) NOT NULL,
    entity_id INT NOT NULL,
    value LONGTEXT NOT NULL,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

INSERT INTO word_replace (data, entity_id, value)
SELECT 'node__body.body_value', entity_id, body_value
FROM node__body
UNION ALL
SELECT 'node__field_smalldesc.field_smalldesc_value', entity_id, field_smalldesc_value
FROM node__field_smalldesc
UNION ALL
SELECT 'block_content__body.body_value', entity_id, body_value
FROM block_content__body;

UPDATE node__body t
JOIN word_replace w ON w.entity_id = t.entity_id
SET t.body_value = w.value
WHERE w.data = 'node__body.body_value';

UPDATE node__field_smalldesc t
JOIN word_replace w ON w.entity_id = t.entity_id
SET t.field_smalldesc_value = w.value
WHERE w.data = 'node__field_smalldesc.field_smalldesc_value';

UPDATE block_content__body t
JOIN word_replace w ON w.entity_id = t.entity_id
SET t.body_value = w.value
WHERE w.data = 'block_content__body.body_value';
