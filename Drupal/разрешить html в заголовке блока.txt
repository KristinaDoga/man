<?
/**
 * Preprocess vars for block.tpl.php.
 */
function THEMENAME_preprocess_block(&$vars) {
  // Allow html title
  if ($vars['block']->title != '<none>' && strpos($vars['block']->title, '<') !== FALSE) {
    $vars['block']->subject = filter_xss($vars['block']->title, array('span', 'div', 'br'));
  }
} ?>
