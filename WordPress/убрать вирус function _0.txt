скрипт, который проходит по всем дочерним файлам php и js, по регулярному выражению найдёт "function _0x.*" и удалит всю строку (см del_0x.php)

Поиск в базе
SELECT * FROM wp_options WHERE option_value LIKE '%function _0x%';
SELECT * FROM wp_posts WHERE post_content LIKE '%function _0x%';
SELECT * FROM wp_postmeta WHERE meta_value LIKE '%function _0x%';
SELECT * FROM wp_wfissues WHERE data LIKE '%function _0x%';


Удалить скрипт
UPDATE wp_posts
SET post_content = LEFT(post_content, LOCATE('<script>function _0x', post_content) - 1)
WHERE post_content LIKE '%function _0x%';
